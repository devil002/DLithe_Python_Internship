```python
def bar(a=100, b=100, c="+"):
    if c == "+":
        return a + b
    elif c == "-":
        return a - b
    elif c == "*":
        return a - b
    elif c == "/":
        return a - b
print(bar())
print(bar(2, 2, "*"))
```
